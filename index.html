const Card3d = require("card3d");

//Vue.js example
import Card3d from "card3d/dist/card3d.js";
Vue.use(Card3d);

//should be a querySelector not querySelectorAll
const card = new Card3d(document.querySelector('.card'), {
    glare: 0.7,
});

//methods
card.stop();
card.start();
